## 问题描述

ubuntu和k8s内，执行操作报错 too many files open 

原因：宿主机的文件句柄数过小，containerd当中可监听的文件句柄数过小

解决方案：

```bash
 #!/bin/bash
 set -e
 
 echo_title() {
   echo -e "\n\033[1;32m>> $1\033[0m"
 }
 
 set_sysctl_param() {
   local key="$1"
   local value="$2"
   local file="/etc/sysctl.conf"
 
   current=$(sysctl -n "$key" 2>/dev/null || echo "")
   if [[ "$current" != "$value" ]]; then
     echo_title "设置 $key = $value"
     sudo sed -i "/^$key[[:space:]]*=.*/d" "$file"
     echo "$key = $value" | sudo tee -a "$file" >/dev/null
     sudo sysctl -w "$key=$value"
   else
     echo "$key 已是 $value，跳过"
   fi
 }
 
 set_limit_conf() {
   local file="/etc/security/limits.conf"
   local pattern="$1"
   grep -q "$pattern" "$file" || echo "$pattern" | sudo tee -a "$file" >/dev/null
 }
 
 set_systemd_conf() {
   local file="$1"
   local key="DefaultLimitNOFILE"
   local value="1048576"
 
   sudo mkdir -p "$(dirname "$file")"
   if grep -q "$key" "$file" 2>/dev/null; then
     sudo sed -i "s/^$key=.*/$key=$value/" "$file"
   else
     echo -e "[Manager]\n$key=$value" | sudo tee "$file" >/dev/null
   fi
 }
 
 add_ulimit_to_bashrc() {
   local user_home
   user_home=$(eval echo "~$1")
   local bashrc="$user_home/.bashrc"
   local line="ulimit -n 1048576"
 
   if grep -qF "$line" "$bashrc"; then
     echo "$1 的 .bashrc 已设置 ulimit，跳过"
   else
     echo_title "设置 $1 的 .bashrc ulimit -n 1048576"
     echo "$line" | sudo tee -a "$bashrc" >/dev/null
   fi
 }
 
 ### 开始执行 ###
 echo_title "1. 设置 limits.conf"
 set_limit_conf "* soft nofile 1048576"
 set_limit_conf "* hard nofile 1048576"
 set_limit_conf "root soft nofile 1048576"
 set_limit_conf "root hard nofile 1048576"
 
 echo_title "2. 设置 systemd 配置"
 set_systemd_conf "/etc/systemd/system.conf.d/99-nofile.conf"
 set_systemd_conf "/etc/systemd/user.conf.d/99-nofile.conf"
 sudo systemctl daemon-reexec
 
 echo_title "3. 设置 sysctl 内核参数"
 set_sysctl_param fs.inotify.max_user_watches 524288
 set_sysctl_param fs.inotify.max_user_instances 1024
 set_sysctl_param fs.inotify.max_queued_events 32768
 set_sysctl_param fs.file-max 2097152
 
 echo_title "4. 设置 bashrc 中的 ulimit -n"
 add_ulimit_to_bashrc "root"
 if [[ "$SUDO_USER" && "$SUDO_USER" != "root" ]]; then
   add_ulimit_to_bashrc "$SUDO_USER"
 fi
 
 echo_title "全部设置完成，请重新登录 shell 或重启容器服务测试效果。"
```
